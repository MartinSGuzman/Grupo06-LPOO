CREATE PROCEDURE dbo.listar_productos_sp
	/*
	(
	@parameter1 int = 5,
	@parameter2 datatype OUTPUT
	)
	*/
AS
	/* SET NOCOUNT ON */
	SELECT
            Prod_Codigo	as 'Codigo',
			Prod_Categoria as 'Categoria',
            Prod_Descripcion as 'Descripcion',
			Prod_Precio 'Precio'
            
		 FROM Producto
